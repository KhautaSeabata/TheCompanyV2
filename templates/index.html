// For now, let's just plot a simple line chart with close prices as a fallback: if (!candleChart) { candleChart = new Chart(ctx, { type: 'line', data: { labels: [], datasets: [{ label: 'Close Price', data: [], borderColor: 'blue', borderWidth: 1, tension: 0.3, }] }, options: { animation: false, scales: { x: { title: { display: true, text: 'Time' }, }, y: { title: { display: true, text: 'Price' } } } } }); } async function fetchCandles(symbol, interval) { const res = await fetch(`/api/candles?symbol=${symbol}&interval=${interval}`); const data = await res.json(); return data; } function updateChart(data) { if (!data || data.length === 0) return; candleChart.data.labels = data.map(c => { const date = new Date(c.epoch * 1000); return date.toLocaleString(); }); candleChart.data.datasets[0].data = data.map(c => c.close); candleChart.update('none'); } async function refresh() { const symbol = document.getElementById("symbol").value; const interval = document.getElementById("interval").value; const candles = await fetchCandles(symbol, interval); updateChart(candles); } document.getElementById("symbol").addEventListener("change", refresh); document.getElementById("interval").addEventListener("change", refresh); // Refresh every 200ms setInterval(refresh, 200); refresh();
